\section*{\faIcon{users} Period A Schedule}
\textit{Status: Already completed candy auction and initial Desmos plotting}

\begin{tcolorbox}[colback=green!10!white,colframe=green!50!black]
\subsection*{Opening \timing{10 minutes}}

\begin{enumerate}[leftmargin=*]
\item \textbf{Vocabulary Review via Blooket} \timing{8-10 min}
    \begin{itemize}
    \item Navigate to: \url{https://dashboard.blooket.com/set/691725bd5c2e794a1c52510b}
    \item Game mode suggestions: Tower Defense or Gold Quest (engaging for review)
    \item Key terms to emphasize: Revenue, Demand, Fixed Cost, Variable Cost, Profit, Break-Even
    \item \note{Let students know these terms will be critical for today's analysis}
    \end{itemize}

\item \textbf{Quick Recap} \timing{2 min}
    \begin{itemize}
    \item ``Last class, we ran our candy auction and plotted demand data''
    \item ``Today, we're converting that to revenue and finding the optimal price''
    \end{itemize}
\end{enumerate}
\end{tcolorbox}

\begin{tcolorbox}[colback=blue!10!white,colframe=blue!50!black]
\subsection*{Main Activity \timing{30-35 minutes}}

\textbf{Distribute:} WKS\_Auction\_to\_Revenue\_Analysis.pdf (+ Spanish version)

\begin{enumerate}[leftmargin=*]
\item \textbf{Part 1-2: Data Transfer} \timing{5 min}
    \begin{itemize}
    \item Students copy their auction data from last class
    \item Fill in Price and Quantity Demanded columns
    \item \note{Walk around to verify everyone has their data}
    \end{itemize}

\item \textbf{Part 3: Calculate Revenue} \timing{5-7 min}
    \begin{itemize}
    \item Guide calculation: Revenue = Price $\times$ Quantity
    \item Have students identify highest revenue point
    \item \tech{Project example calculation on board}
    \end{itemize}

\item \textbf{Part 4: Desmos Revenue Graph} \timing{8-10 min}
    \begin{itemize}
    \item Open new Desmos table (or modify existing)
    \item Enter Price ($x_1$) and Revenue ($y_1$) data
    \item Students should see parabolic shape forming
    \item \note{Help struggling students with Desmos interface}
    \end{itemize}

\item \textbf{Part 5: Quadratic Regression} \timing{7-8 min}
    \begin{itemize}
    \item Type in Desmos: \texttt{y\_1 \textasciitilde{} a*x\_1\^{}2 + b*x\_1 + c}
    \item Record equation as $I(p) = ap^2 + bp + c$
    \item Discuss why $a$ is negative (parabola opens down)
    \end{itemize}

\item \textbf{Part 6: Find Vertex} \timing{5-7 min}
    \begin{itemize}
    \item Click on maximum point in Desmos (graphical method)
    \item Calculate using $x = -\frac{b}{2a}$ (algebraic method)
    \item Verify both methods give same answer
    \item Interpret as ``optimal price for maximum revenue''
    \end{itemize}
\end{enumerate}
\end{tcolorbox}

\begin{tcolorbox}[colback=gray!10!white,colframe=gray!50!black]

\section*{\faIcon{lightbulb} Teaching Tips \& Troubleshooting}

\begin{tcolorbox}[colback=yellow!10!white,colframe=yellow!50!black,title=Common Issues \& Solutions]
\begin{itemize}[leftmargin=*]
\item \textbf{Desmos won't do regression}
    \begin{itemize}
    \item Make sure to use tilde (\textasciitilde) not equals
    \item Check that $x_1$ and $y_1$ match table column names
    \item Try: \texttt{y1 \textasciitilde{} a*x1\^{}2 + b*x1 + c}
    \end{itemize}

\item \textbf{Students getting different vertex values}
    \begin{itemize}
    \item Rounding differences are OK
    \item Emphasize both methods (graphical \& algebraic) should be close
    \item Use this as teaching moment about precision
    \end{itemize}

\item \textbf{Negative revenue values}
    \begin{itemize}
    \item Check multiplication: Price $\times$ Quantity
    \item Revenue should always be positive
    \item Common error: subtracting instead of multiplying
    \end{itemize}

\item \textbf{Spanish-speaking students need help}
    \begin{itemize}
    \item Distribute Spanish version alongside English
    \item Pair with bilingual student if possible
    \item Key terms: Ingreso (Revenue), Demanda (Demand), Precio (Price)
    \end{itemize}
\end{itemize}
\end{tcolorbox}

\begin{tcolorbox}[colback=green!10!white,colframe=green!50!black,title=Key Questions to Ask Students]
\begin{itemize}[leftmargin=*]
\item ``Why does revenue go down if price is too high OR too low?''
\item ``What does the vertex represent in business terms?''
\item ``How would this change if we cared about profit instead of revenue?''
\item ``Could we use this same method for the school cafeteria?''
\end{itemize}
\end{tcolorbox}
